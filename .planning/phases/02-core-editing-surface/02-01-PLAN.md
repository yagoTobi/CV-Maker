    **EditableField.tsx** -- Named export `EditableField`. Props interface:
    ```typescript
    interface EditableFieldProps {
      value: string;
      fieldPath: string;
      onFieldChange: (path: string, value: string) => void;
      placeholder?: string;
      tag?: 'span' | 'div' | 'h1' | 'h2' | 'p';
      className?: string;
      multiline?: boolean;
      onInput?: () => void;
      onKeyDown?: (e: React.KeyboardEvent) => void;
    }
    ```

    Implementation pattern (per D-04, "uncontrolled while focused, controlled while blurred"):
    - `useRef<HTMLElement>(null)` for DOM element
    - `isFocused = useRef(false)` to track focus state
    - `useEffect` syncs `value` prop to `ref.current.textContent` ONLY when `isFocused.current === false` AND `ref.current.textContent !== value`
    - `handleFocus`: sets `isFocused.current = true`
    - `handleBlur`: sets `isFocused.current = false`, reads `ref.current.textContent`, calls `onFieldChange(fieldPath, newValue)` only if value changed
    - `handleKeyDown`: Enter on non-multiline -> `e.preventDefault(); ref.current.blur()`. Escape -> revert textContent to `value`, then blur. Delegates to `onKeyDown` prop if provided.
    - `handleInput`: calls `onInput?.()` for auto-save debounce trigger
    - Renders `<Tag ref={ref} contentEditable="plaintext-only" suppressContentEditableWarning data-field-path={fieldPath} data-placeholder={placeholder} className={className} onFocus={handleFocus} onBlur={handleBlur} onInput={handleInput} onKeyDown={handleKeyDown} />`
    - Use `React.forwardRef` so parent can access the DOM element (needed by EditableBulletList for focus management)
    - Track IME composition with `compositionstart`/`compositionend` events; gate `onInput` calls during active composition (per Pitfall 4 in RESEARCH.md)

    **useDirectEditor.ts** -- Named export `useDirectEditor`. Implementation:

    ```typescript
    import { useCallback } from 'react';
    import { useCVContext } from '../../../contexts/CVContext';
    import { generateId } from '../../../utils/idHelpers';
    import type { CVFormData, BulletItem } from '../../../types';

    function parsePath(path: string): (string | number)[] {
      const segments: (string | number)[] = [];
      const re = /([^.[]+)|\[(\d+)\]/g;
      let m: RegExpExecArray | null;
      while ((m = re.exec(path)) !== null) {
        if (m[1] !== undefined) segments.push(m[1]);
        else if (m[2] !== undefined) segments.push(Number(m[2]));
      }
      return segments;
    }

    function setAtPath(obj: Record<string, unknown>, path: string, value: unknown): void {
      // Same logic as formDataPatch.ts setAtPath — preserves BulletItem/SkillItem IDs
    }

    function getAtPath(obj: Record<string, unknown>, path: string): unknown {
      const segs = parsePath(path);
      let cur: unknown = obj;
      for (const seg of segs) {
        if (cur == null) return undefined;
        cur = (cur as Record<string, unknown>)[seg as string];
      }
      return cur;
    }
    ```
