const { setActiveVersion } = useCVContext();

const handleSaveBase = useCallback(async () => {
  if (!baseName.trim()) return;
  setSavingBase(true);
  const { texContent } = await api.generateLatex(formData);
  const saved = await api.saveVersion({
    name: baseName.trim(),
    templateId: formData.templateId,
    texContent: texContent || '',
    formData,
  });
  if (saved) {
    // Load the saved version as the active version
    const fullVersion = await api.getVersion(saved.id);
    if (fullVersion) {
      setActiveVersion(fullVersion);
    }
    setTier1Complete(true);
    setActiveTier(2);
  }
  setSavingBase(false);
}, [baseName, formData, setActiveVersion]);

**Handle Tier 2 analyze (port from ApplyToJobScreen.handleAnalyze):**
```typescript
const handleAnalyze = useCallback(async () => {
  if (!jobDescription.trim()) return;
  setAnalyzing(true);
  const { texContent } = await api.generateLatex(formData);
  if (!texContent) { setAnalyzing(false); return; }
  const analysis = await api.getMatchAnalysis(texContent, jobDescription, companyName);
  if (analysis) {
    setMatchAnalysis(analysis);
    setBaselineScore(analysis.match_score);
    setTier2Complete(true);
    setActiveTier(3);
  }
  setAnalyzing(false);
  // Initialize preview and pre-fetch tailor suggestions
  setPreviewFormData(formData);
  tailor.fetchSuggestions(formData, jobDescription, companyName, roleName);
}, [formData, jobDescription, companyName, roleName, tailor]);
```

const navigate = useNavigate();

const handleSaveTailored = useCallback(async () => {
  const currentVersion = activeVersion;
  if (!currentVersion || !previewFormData) return;
  setSaving(true);
  const { texContent } = await api.generateLatex(previewFormData);
  if (!texContent) { setSaving(false); return; }
  const finalAnalysis = await api.getMatchAnalysis(texContent, jobDescription, companyName);
  const score = finalAnalysis?.match_score ?? tailor.estimatedCurrentScore;
  const versionName = [companyName, roleName].filter(Boolean).join(' ') || 'Job Application';
  const saved = await api.saveVersion({
    name: versionName,
    templateId: currentVersion.templateId,
    texContent,
    formData: previewFormData,
    jobDescription,
    companyName,
    role: roleName,
    matchScore: score,
    baselineMatchScore: baselineScore,
    parentVersionId: currentVersion.id,
  });
  if (saved) {
    navigate('/dashboard', { state: { baseId: currentVersion.id } });
  }
  setSaving(false);
}, [activeVersion, previewFormData, jobDescription, companyName, roleName, baselineScore, tailor.estimatedCurrentScore, navigate]);

**IMPORTANT implementation notes:**

1. Tier bodies use CSS expand/collapse (NOT conditional rendering) so state is preserved when collapsed (RESEARCH.md Pitfall 3). The pattern:
   - `.tierBody` has `max-height: 0; overflow: hidden; transition: max-height 200ms ease`
   - `.tierBodyOpen` has `max-height: 800px` (large enough for any content)
   - `.tierBodyInner` wraps the actual content

2. ChangePanel inside Tier 3 is embedded inline -- NOT as a fixed-position side panel. Override ChangePanel's fixed positioning by wrapping it or setting the parent's position context. Since ChangePanel uses `position: fixed`, wrap it in a div with `className={styles.changePanelEmbed}` and use CSS to override:
```css
.changePanelEmbed :global(.panel) {
  position: static;
  width: 100%;
  height: auto;
  overflow: visible;
  border-left: none;
  padding: 0;
  transform: none;
}
```
